#!/bin/bash - Указываем интерпрететор bash. С этого начинается каждый скрипт

#Проверка, запущен ли скрипт от root. Если нет, то выходим.
if (($EUID !=0)); then
     echo Script must be run by root.
     exit
fi

#Ввод переменной a с клавиатуры
read a

#Вывод переменной a на экран
echo $a

#Обращение к переменной a производится следующим образом:  $a

# Параметры скрипта и обращение к ним
# Параметры пишутся после скрипта. Пример ./script.sh 1й_параметр 2й_параметр 3й_параметр и т.д.
#Обращение к ним происходит следующим образом:
$0 - Названия скрипта
$1 - Первый параметр
$2 - Второй параметр
$3 - Третий параметр

#Пример:
echo $1 - Вывести первый параметр скрипта
#Использование Case:
echo; echo "Нажмите клавишу и затем клавишу Return."
read Keypress #Ввод переменной Keypress

case "$Keypress" in
  [a-z]   ) echo "буква в нижнем регистре";;
  [A-Z]   ) echo "Буква в верхнем регистре";;
  [0-9]   ) echo "Цифра";;
  *       ) echo "Знак пунктуации, пробел или что-то другое";;
esac  # Допускается указыватль диапазоны символов в [квадратных скобках].

#Yes/No

read -r -p "Да или нет. [Yes/No] " response
case "$response" in
    [yY][eE][sS]|[yY]|[Дд]|[Аа]) 
        echo "Выбрано да."
        ;;
    *)
        echo "Выбрано нет."
        ;;
esac

#Проверка пустого параметра. $1 - первый параметр скрипта.
if [ -z "$1" ] ;then
echo 'Ничего не введено.'
else
echo Введено $1
fi
